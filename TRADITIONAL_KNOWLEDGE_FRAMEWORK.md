```python
# src/knowledge_graph/traditional_knowledge_graph.py

from neo4j import GraphDatabase

class TraditionalKnowledgeGraph:
    """
    Graph database for interconnected traditional knowledge
    Not a spreadsheet - a living web of relationships
    """
    
    def __init__(self):
        self.driver = GraphDatabase.driver("bolt://localhost:7687")
    
    def add_observation(self, observation):
        """
        Add elder observation with all relationships
        """
        with self.driver.session() as session:
            # Create node
            session.run("""
                CREATE (o:Observation {
                    id: $id,
                    description: $desc,
                    method: $method,
                    date: $date
                })
                """, 
                id=observation['id'],
                desc=observation['description'],
                method=observation['method'],
                date=observation['date']
            )
            
            # Create relationships
            # To elder
            session.run("""
                MATCH (o:Observation {id: $obs_id})
                MATCH (e:Elder {name: $elder})
                CREATE (o)-[:OBSERVED_BY]->(e)
                """,
                obs_id=observation['id'],
                elder=observation['elder']
            )
            
            # To related phenomena
            for related in observation['related_to']:
                session.run("""
                    MATCH (o:Observation {id: $obs_id})
                    MERGE (r:Phenomenon {name: $phenomenon})
                    CREATE (o)-[:INDICATES]->(r)
                    """,
                    obs_id=observation['id'],
                    phenomenon=related
                )
    
    def find_resonance(self, query):
        """
        Find knowledge that resonates across multiple domains
        """
        with self.driver.session() as session:
            result = session.run("""
                MATCH (o:Observation)-[:INDICATES]->(p:Phenomenon)
                WHERE p.name CONTAINS $query
                
                // Find all related knowledge
                MATCH (o)-[:OBSERVED_BY]->(e:Elder)
                MATCH (o)-[:VALIDATED_BY]->(v:Validation)
                MATCH (o)-[:CONNECTS_TO]->(other:Observation)
                
                // Calculate resonance score
                WITH o, e, v, COUNT(other) as connections
                WHERE connections > 2  // Multiple confirming observations
                
                RETURN o, e, v, connections
                ORDER BY connections DESC
                """,
                query=query
            )
            return result.data()
```
